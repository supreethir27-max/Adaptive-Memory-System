<script>
const loginPage=document.getElementById('loginPage');
const dashboardPage=document.getElementById('dashboardPage');
const quizPage=document.getElementById('quizPage');
const loginBtn=document.getElementById('loginBtn');
const fileInput=document.getElementById('fileInput');
const manualContent=document.getElementById('manualContent');
const summarizeBtn=document.getElementById('summarizeBtn');
const summaryOutput=document.getElementById('summaryOutput');
const keyPoints=document.getElementById('keyPoints');
const lessonName=document.getElementById('lessonName');
const lessonDuration=document.getElementById('lessonDuration');
const addLessonBtn=document.getElementById('addLessonBtn');
const lessonsList=document.getElementById('lessonsList');
const generateQuizBtn=document.getElementById('generateQuizBtn');
const quizOutput=document.getElementById('quizOutput');

let lessons=[]; let timers={}; let activeAlarm=null;

loginBtn.onclick=()=>{
  if(!userName.value || !userEmail.value || !userPassword.value) return alert('Fill all fields');
  loginPage.classList.remove('show'); dashboardPage.classList.add('show');
}

window.showQuizPage=()=>{dashboardPage.classList.remove('show'); quizPage.classList.add('show');}
window.showDashboard=()=>{quizPage.classList.remove('show'); dashboardPage.classList.add('show');}

fileInput.addEventListener('change', async e=>{
 const file=e.target.files[0]; if(!file) return;
 if(file.type==='application/pdf'){
  const buffer=await file.arrayBuffer();
  const pdf=await pdfjsLib.getDocument(new Uint8Array(buffer)).promise;
  let text='';
  for(let i=1;i<=pdf.numPages;i++){
   const page=await pdf.getPage(i); const content=await page.getTextContent();
   text += content.items.map(it=>it.str).join(' ') + '\n';
  }
  manualContent.value=text;
 } else manualContent.value=await file.text();
});

summarizeBtn.onclick=()=>{
 const text=manualContent.value.trim(); if(!text) return;
 const s=text.split(/[.!?]/).map(x=>x.trim()).filter(Boolean);
 summaryOutput.innerHTML='<b>Summary:</b><br>'+s.slice(0,3).join('. ')+'.';
 keyPoints.innerHTML='<h3>Key Points</h3>'+s.slice(0,5).map((x,i)=>`<div class="list-item">${i+1}. ${x}</div>`).join('');
}

function startAlarm(name){
 const ctx=new (window.AudioContext||window.webkitAudioContext)();
 const osc=ctx.createOscillator(); osc.frequency.value=880; osc.connect(ctx.destination); osc.start();
 activeAlarm={ctx,osc};
 const stop=confirm('🔔 Time to revise '+name+'\nPress OK to stop alarm');
 if(stop){ osc.stop(); ctx.close(); activeAlarm=null; }
}

function renderLessons(){
 lessonsList.innerHTML=lessons.map(l=>`<div class="list-item"><b>${l.name}</b><br>${l.duration} min
 <br><button class="secondary" onclick="deleteLesson(${l.id})">🗑 Delete</button></div>`).join('') || 'No lessons';
}

window.deleteLesson=(id)=>{clearTimeout(timers[id]); lessons=lessons.filter(x=>x.id!==id); renderLessons();}

addLessonBtn.onclick=()=>{
 const name=lessonName.value.trim(); const duration=parseInt(lessonDuration.value);
 if(!name||!duration) return alert('Enter lesson and time');
 const lesson={id:Date.now(),name,duration}; lessons.push(lesson); renderLessons();
 timers[lesson.id]=setTimeout(()=>startAlarm(lesson.name), duration*60000);
 lessonName.value=''; lessonDuration.value='';
}

generateQuizBtn.onclick=()=>{
 const text=manualContent.value.trim();
 if(!text) return alert('Upload content first');
 const lines=text.split(/[.!?]/).map(x=>x.trim()).filter(x=>x.length>20).slice(0,5);
 quizOutput.innerHTML='<h3>Generated Questions</h3>'+lines.map((l,i)=>`<div class="list-item">Q${i+1}: Explain ${l.split(' ').slice(0,6).join(' ')}?</div>`).join('');
}
